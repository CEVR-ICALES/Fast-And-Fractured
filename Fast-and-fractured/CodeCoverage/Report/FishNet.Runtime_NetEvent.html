 734        private int GetNextPeerId()
 735        {
 736            /* Build a small queue so that when fetching
 737            * an id its unlikely to be one that was just returned. This adds
 738            * a buffer to applications which may be using the Id to identity
 739            * clients in their code, but may need time to dispose of objects
 740            * associated with the Id. */
 741            const int addCount = 5000;
 742            const int halfCount = (addCount / 2);
 743            if (_peerIds.Count < halfCount)
 744            {
 745                //At most how many can be added.
 746                const int maxIdValue = int.MaxValue;
 747                long addable = Math.Min(addCount, (maxIdValue - _nextPeerId));
 748
 749                if (addable == 0)
 750                {
 751                    NetDebug.Write(NetLogLevel.Error, $"[NM] PeerIds have exceeded the maximum value of {maxIdValue}. A 
 752                    return 0;
 753                }

 755                for (int i = 0; i < addCount; i++)
 756                    _peerIds.Enqueue(_nextPeerId++);
 757            }
 758
 759            return (_peerIds.TryDequeue(out int result)) ? result : _nextPeerId++;
 760        }
